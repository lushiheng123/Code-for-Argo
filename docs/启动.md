# 在 WSL 终端中执行（保持窗口打开）
# 转发 ArgoCD Web UI
kubectl -n argocd port-forward svc/argocd-server 8080:80 &

# 转发 Grafana(admin/admin123)
kubectl -n monitoring port-forward svc/monitoring-grafana 3000:80 &

# 转发 Prometheus
kubectl -n monitoring port-forward svc/monitoring-prometheus-server 9090:80 &

# 转发 Alertmanager
kubectl -n monitoring port-forward svc/monitoring-alertmanager 9093:9093 &

# 验证端口是否在监听
ss -tlnp | grep -E "8080|3000|9090|9093"
